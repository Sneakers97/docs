---
title: 'Get Post'
api: 'GET https://api.infinipost.co/v1/posts/{id}'
description: 'Retrieve details for a specific post'
---

## Path Parameters

<ParamField path="id" type="string" required>
  The unique post identifier.
</ParamField>

## Response

<ResponseField name="id" type="string">
  Post identifier.
</ResponseField>

<ResponseField name="accountId" type="string">
  Associated account ID.
</ResponseField>

<ResponseField name="platform" type="string">
  Platform (`tiktok` or `instagram`).
</ResponseField>

<ResponseField name="contentType" type="string">
  `video` or `slideshow`.
</ResponseField>

<ResponseField name="caption" type="string">
  Post caption.
</ResponseField>

<ResponseField name="status" type="string">
  Status: `scheduled`, `posted`, `failed`.
</ResponseField>

<ResponseField name="stats" type="object">
  Post performance metrics.
  
  <Expandable title="properties">
    <ResponseField name="views" type="number">
      Number of views.
    </ResponseField>
    
    <ResponseField name="likes" type="number">
      Number of likes.
    </ResponseField>
    
    <ResponseField name="comments" type="number">
      Number of comments.
    </ResponseField>
    
    <ResponseField name="shares" type="number">
      Number of shares.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="postedAt" type="string">
  ISO 8601 timestamp of when posted (if status is `posted`).
</ResponseField>

<RequestExample>

```bash cURL
curl https://api.infinipost.co/v1/posts/post_abc123 \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch('https://api.infinipost.co/v1/posts/post_abc123', {
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  }
});

const data = await response.json();
```

</RequestExample>
