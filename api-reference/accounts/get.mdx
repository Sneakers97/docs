---
title: 'Get Account'
api: 'GET https://api.infinipost.co/v1/accounts/{id}'
description: 'Retrieve details for a specific account'
---

## Path Parameters

<ParamField path="id" type="string" required>
  The unique account identifier.
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the account.
</ResponseField>

<ResponseField name="platform" type="string">
  Platform: `tiktok` or `instagram`.
</ResponseField>

<ResponseField name="name" type="string">
  Account display name.
</ResponseField>

<ResponseField name="username" type="string">
  Account username.
</ResponseField>

<ResponseField name="bio" type="string">
  Account bio/description.
</ResponseField>

<ResponseField name="profilePictureUrl" type="string">
  URL of the profile picture.
</ResponseField>

<ResponseField name="status" type="string">
  Current status: `pending`, `active`, `warmup`, or `failed`.
</ResponseField>

<ResponseField name="stats" type="object">
  Account statistics.
  
  <Expandable title="properties">
    <ResponseField name="followers" type="number">
      Number of followers.
    </ResponseField>
    
    <ResponseField name="following" type="number">
      Number of accounts following.
    </ResponseField>
    
    <ResponseField name="posts" type="number">
      Number of posts.
    </ResponseField>
    
    <ResponseField name="totalViews" type="number">
      Total video views.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="warmupStatus" type="object">
  Warmup details.
</ResponseField>

<ResponseField name="createdAt" type="string">
  ISO 8601 timestamp.
</ResponseField>

<RequestExample>

```bash cURL
curl https://api.infinipost.co/v1/accounts/acc_1a2b3c4d5e \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```javascript JavaScript
const response = await fetch('https://api.infinipost.co/v1/accounts/acc_1a2b3c4d5e', {
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://api.infinipost.co/v1/accounts/acc_1a2b3c4d5e',
    headers={'Authorization': 'Bearer YOUR_API_KEY'}
)

print(response.json())
```
